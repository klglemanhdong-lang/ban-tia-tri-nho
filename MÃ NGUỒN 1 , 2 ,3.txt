    <script>
        const bgMusic = new Audio('nhac-nen.mp3'); bgMusic.loop = true; bgMusic.volume = 0.5;
        const reloadSound = new Audio('nap-dan.mp3');
        const boomSound = new Audio('bum.mp3');
        const wrongSound = new Audio('sai.mp3');
        const victorySound = new Audio('chien-thang.mp3');

        const gameData = [
            { id: 0, label: "00", text: "Quả trứng" }, { id: 1, label: "01", text: "Cây dù" },
            { id: 2, label: "02", text: "Con vịt" }, { id: 3, label: "03", text: "Trái tim" },
            { id: 4, label: "04", text: "Cái ghế" }, { id: 5, label: "05", text: "Quả táo" },
            { id: 6, label: "06", text: "Vỏ ốc" }, { id: 7, label: "07", text: "Cây rìu" },
            { id: 8, label: "08", text: "Người tuyết" }, { id: 9, label: "09", text: "Hươu cao cổ" },
            { id: 10, label: "10", text: "Con heo" }
        ];

        let selectedNumberId = null;
        let selectedNumberElement = null;
        let matchedPairs = 0;
        let timeLeft = 60;
        let timerInterval;
        let isAnimating = false;
        let isMusicPlaying = false;

        function shuffle(array) {
            let currentIndex = array.length, randomIndex;
            while (currentIndex !== 0) {
                randomIndex = Math.floor(Math.random() * currentIndex);
                currentIndex--;
                [array[currentIndex], array[randomIndex]] = [array[randomIndex], array[currentIndex]];
            }
            return array;
        }

        function startGame() {
            selectedNumberId = null;
            selectedNumberElement = null;
            matchedPairs = 0;
            timeLeft = 60;
            isAnimating = false;
            isMusicPlaying = false; 

            document.getElementById('rule-modal').classList.remove('active');
            document.getElementById('lose-modal').classList.remove('active');
            document.getElementById('win-modal').classList.remove('active');

            renderBoard();
            clearInterval(timerInterval);
            timerInterval = setInterval(updateTimer, 1000);
            
            bgMusic.play();
            isMusicPlaying = true;
        }

        window.addEventListener('keydown', function(event) {
            if (event.key === 't' || event.key === 'T') {
                handleWin(); 
            }
            if (event.key === 'l' || event.key === 'L') {
                handleLose(); 
            }
        });

        function renderBoard() {
            const topRow = document.getElementById('top-row');
            const bottomRow = document.getElementById('bottom-row');
            topRow.innerHTML = ''; bottomRow.innerHTML = '';

            gameData.forEach(item => {
                const numCard = document.createElement('div');
                numCard.classList.add('card', 'number-card');
                numCard.innerText = item.label; 
                numCard.onclick = () => handleNumberClick(item.id, numCard);
                topRow.appendChild(numCard);
            });

            const shuffledData = shuffle([...gameData]);
            shuffledData.forEach(item => {
                const imgCard = document.createElement('div');
                imgCard.classList.add('card', 'image-card');
                imgCard.innerHTML = `
                    <img src="${item.id === 10 ? '10.png' : item.label + '.png'}" alt="${item.text}">
                    <div class="image-name">${item.text}</div>
                `;
                imgCard.onclick = () => handleImageClick(item.id, imgCard);
                bottomRow.appendChild(imgCard);
            });
        }

        function handleNumberClick(id, element) {
            if (isAnimating || element.classList.contains('hidden-card')) return;
            reloadSound.currentTime = 0; reloadSound.play();
            if (selectedNumberElement) selectedNumberElement.classList.remove('selected', 'broken');
            selectedNumberId = id;
            selectedNumberElement = element;
            element.classList.add('selected');
        }

        function handleImageClick(id, element) {
            if (isAnimating || selectedNumberId === null || element.classList.contains('hidden-card')) return;
            isAnimating = true;

            if (selectedNumberId === id) {
                boomSound.currentTime = 0; boomSound.play();
                element.classList.add('burning');
                selectedNumberElement.classList.remove('selected');
                
                setTimeout(() => {
                    element.classList.add('hidden-card');
                    selectedNumberElement.classList.add('hidden-card');
                    matchedPairs++;
                    resetSelection();
                    if (matchedPairs === gameData.length) handleWin();
                }, 500);
            } else {
                wrongSound.currentTime = 0; wrongSound.play();
                element.classList.add('shake');
                selectedNumberElement.classList.remove('selected');
                selectedNumberElement.classList.add('broken');

                setTimeout(() => {
                    element.classList.remove('shake');
                    selectedNumberElement.classList.remove('broken');
                    resetSelection();
                }, 800);
            }
        }

        function resetSelection() {
            selectedNumberId = null;
            selectedNumberElement = null;
            isAnimating = false;
        }

        function updateTimer() {
            timeLeft--;
            document.getElementById('time-left').innerText = timeLeft;
            if (timeLeft <= 0) {
                handleLose();
            }
        }

        function handleWin() {
            clearInterval(timerInterval);
            bgMusic.pause();
            victorySound.play();
            document.getElementById('win-modal').classList.add('active');
        }

        function handleLose() {
            clearInterval(timerInterval);
            bgMusic.pause();
            document.getElementById('lose-modal').classList.add('active');
        }
    </script>

        const bgMusic = new Audio('nhac-nen.mp3'); bgMusic.loop = true; bgMusic.volume = 0.5;

        ];

        let lives = 5; let score = 0; let gameActive = false;
        let meteorQueue = []; let spawnInterval;
        let fallSpeed = 9; let spawnTime = 2500;

        const base = document.getElementById('base');
        gameData.forEach(item => {
            const wrapper = document.createElement('div');
            wrapper.className = 'turret-wrapper';
            wrapper.innerHTML = `<div class="cannon" id="cannon-${item.id}">🗼</div>
                                <button class="btn-shoot" onclick="shoot(${item.id})">${item.label}</button>`;
            base.appendChild(wrapper);

        function startGame() {
            document.getElementById('rule-modal').classList.remove('active');
            gameActive = true; bgMusic.play();
            startSpawning();
        }

        // PHÍM TẮT ĐỂ TEST (T & L)
        window.addEventListener('keydown', function(event) {
            if (event.key === 't' || event.key === 'T') {
                handleWin(); 
            }
            if (event.key === 'l' || event.key === 'L') {
                handleLose(); 
            }
        });

        function startSpawning() {
            clearInterval(spawnInterval);
            spawnInterval = setInterval(spawnMeteor, spawnTime);
        }

        function spawnMeteor() {
            if (!gameActive) return;
            if (meteorQueue.length === 0) meteorQueue = [...gameData].sort(() => Math.random() - 0.5);
            const item = meteorQueue.pop();
            const meteor = document.createElement('div');
            meteor.className = 'meteor';
            meteor.dataset.id = item.id;
            meteor.style.left = (Math.random() * 80 + 10) + '%';
            meteor.style.animationDuration = fallSpeed + 's';
            meteor.innerHTML = `<img src="${item.id === 10 ? '10.png' : item.label + '.png'}"><div class="meteor-text">${item.text}</div>`;
            document.getElementById('sky').appendChild(meteor);
            
            meteor.addEventListener('animationend', () => {
                if (document.body.contains(meteor)) { meteor.remove(); takeDamage(); }
            });
        }

        function shoot(id) {
            if (!gameActive) return;
            const cannon = document.getElementById('cannon-' + id);
            const cRect = cannon.getBoundingClientRect();
            reloadSound.currentTime = 0; reloadSound.play();

            const targets = Array.from(document.getElementsByClassName('meteor')).filter(m => m.dataset.id == id);
            if (targets.length > 0) {
                const target = targets.reduce((l, c) => c.getBoundingClientRect().top > l.getBoundingClientRect().top ? c : l);
                const tRect = target.getBoundingClientRect();
                drawLaser(cRect.left + cRect.width/2, cRect.top, tRect.left + tRect.width/2, tRect.top + tRect.height/2);
                target.remove();
                boomSound.currentTime = 0; boomSound.play();
                score++; document.getElementById('score').innerText = score;
                checkDifficulty();
            }
        }

        function checkDifficulty() {
            if (score >= 30) {
                handleWin();
            } else if (score === 20) {
                fallSpeed = 4.5; spawnTime = 1200; 
                document.getElementById('level-indicator').innerText = "⭐⭐⭐"; 
                startSpawning();
            } else if (score === 10) {
                fallSpeed = 6.5; spawnTime = 1800; 
                document.getElementById('level-indicator').innerText = "⭐⭐"; 
                startSpawning();
            }
        }

        function drawLaser(x1, y1, x2, y2) {
            const len = Math.hypot(x2-x1, y2-y1);
            const ang = Math.atan2(y2-y1, x2-x1) * 180 / Math.PI;
            const l = document.createElement('div');
            l.className = 'laser'; l.style.width = len + 'px'; l.style.left = x1 + 'px';
            l.style.top = y1 + 'px'; l.style.transform = `rotate(${ang}deg)`;
            document.body.appendChild(l); setTimeout(() => l.remove(), 250);
        }

        function takeDamage() {
            wrongSound.play(); lives--;
            let h = ''; for(let i=0; i<lives; i++) h += '❤️';
            document.getElementById('lives').innerText = h || '💀';
            if (lives <= 0) handleLose();
        }

        function handleWin() { gameActive = false; clearInterval(spawnInterval); bgMusic.pause(); victorySound.play(); document.getElementById('win-modal').classList.add('active'); }
        function handleLose() { gameActive = false; clearInterval(spawnInterval); bgMusic.pause(); document.getElementById('lose-modal').classList.add('active'); }
    </script>

    <script>
        const bgMusic = new Audio('nhac-nen.mp3'); bgMusic.loop = true; bgMusic.volume = 0.5;
        const reloadSound = new Audio('nap-dan.mp3');
        const boomSound = new Audio('bum.mp3');
        const wrongSound = new Audio('sai.mp3');
        const victorySound = new Audio('chien-thang.mp3');

        const gameData = [
            { id: 0, label: "00", text: "Quả trứng" }, { id: 1, label: "01", text: "Cây dù" },
            { id: 2, label: "02", text: "Con vịt" }, { id: 3, label: "03", text: "Trái tim" },
            { id: 4, label: "04", text: "Cái ghế" }, { id: 5, label: "05", text: "Quả táo" },
            { id: 6, label: "06", text: "Vỏ ốc" }, { id: 7, label: "07", text: "Cây rìu" },
            { id: 8, label: "08", text: "Người tuyết" }, { id: 9, label: "09", text: "Hươu cao cổ" },
            { id: 10, label: "10", text: "Con heo" }
        ];

        let lives = 5; let score = 0; let gameActive = false;
        let spawnTimeout; let startTime;
        let currentSpeed = 8; 

        const panel = document.getElementById('control-panel');
        gameData.forEach(item => {
            const btn = document.createElement('button');
            btn.className = 'btn-target'; btn.innerText = item.label;
            btn.onclick = () => shoot(item.id);
            panel.appendChild(btn);
        });

        function startGame() {
            document.getElementById('rule-modal').classList.remove('active');
            gameActive = true; 
            startTime = Date.now(); 
            bgMusic.play();
            startSpawning();
        }

        function startSpawning() {
            const cycle = () => {
                if(!gameActive) return;
                const currentMeteors = document.getElementsByClassName('meteor').length;
                if (currentMeteors < 3) spawnMeteor();
                let delay = score > 30 ? 1500 : (score > 20 ? 2000 : 2500);
                currentSpeed = score > 30 ? 5.5 : (score > 15 ? 6.5 : 8);
                spawnTimeout = setTimeout(cycle, delay);
            };
            cycle();
        }

        function spawnMeteor() {
            const item = gameData[Math.floor(Math.random() * gameData.length)];
            const meteor = document.createElement('div');
            meteor.className = 'meteor'; meteor.dataset.id = item.id;
            meteor.dataset.hit = "false"; // Để không bị trừ mạng khi đang nổ đặc biệt
            const fromLeft = Math.random() > 0.5;
            meteor.style.top = (Math.random() * 60 + 5) + '%';
            meteor.style.animation = (fromLeft ? 'flyLeftToRight' : 'flyRightToLeft') + ` ${currentSpeed}s linear forwards`;
            meteor.innerHTML = `<img src="${item.id === 10 ? '10.png' : item.label + '.png'}"><div class="meteor-text">${item.text}</div>`;
            document.getElementById('space').appendChild(meteor);
            meteor.addEventListener('animationend', () => {
                if(document.body.contains(meteor)) { 
                    if(meteor.dataset.hit === "false") {
                        meteor.remove(); if(gameActive) takeDamage(); 
                    } else {
                        meteor.remove();
                    }
                }
            });
        }

        function shoot(id) {
            if(!gameActive) return;
            reloadSound.currentTime = 0; reloadSound.play();
            const targets = Array.from(document.getElementsByClassName('meteor')).filter(m => m.dataset.id == id && m.dataset.hit === "false");
            
            if(targets.length > 0) {
                const target = targets[0];
                target.dataset.hit = "true"; // Đánh dấu đã bắn trúng
                score++;
                document.getElementById('score').innerText = score;
                boomSound.currentTime = 0; boomSound.play();

                // Kiểm tra mỗi 5 lần bắn trúng (5, 10, 15...)
                if (score % 5 === 0) {
                    // Chốt vị trí để không bị nhảy sang trái
                    const rect = target.getBoundingClientRect();
                    target.style.left = rect.left + 'px';
                    target.style.animation = 'none'; 
                    
                    // Hiện hiệu ứng +5
                    const plus = document.createElement('div');
                    plus.className = 'bonus-plus';
                    plus.innerText = '+5';
                    plus.style.left = (rect.left + rect.width/2 - 20) + 'px';
                    plus.style.top = rect.top + 'px';
                    document.body.appendChild(plus);
                    setTimeout(() => plus.remove(), 1000);

                    target.classList.add('special-effect');
                    setTimeout(() => { if(target.parentNode) target.remove(); }, 800);
                } else {
                    target.style.pointerEvents = 'none';
                    target.innerHTML = '<span style="font-size:50px">💥</span>';
                    setTimeout(() => { if(target.parentNode) target.remove(); }, 200);
                }

                if(score >= 40) handleWin();
            }
        }

        function takeDamage() {
            if(!gameActive) return;
            wrongSound.play(); lives--;
            let h = ''; for(let i=0; i<lives; i++) h += '❤️';
            document.getElementById('lives').innerText = h || '💀';
            if(lives <= 0) handleLose();
        }

        function handleWin() {
            gameActive = false;
            clearTimeout(spawnTimeout);
            bgMusic.pause();
            victorySound.play();
            const timeTaken = Math.floor((Date.now() - startTime) / 1000);

            document.getElementById('win-modal').innerHTML = `
                <style>
                    @keyframes bronzeBadgeZoom {
                        0% { transform: scale(1); }
                        50% { transform: scale(1.15); } 
                        100% { transform: scale(1); }
                    }
                    .animated-badge {
                        animation: bronzeBadgeZoom 2s ease-in-out infinite;
                    }
                </style>
                <img src="1dong.png" class="animated-badge" style="width: 300px; margin-bottom: 20px; filter: drop-shadow(0 0 20px #cd7f32);">
                <h2 style="color: #FFD700; font-size: 50px; text-shadow: 0 0 20px #FFD700;">Tuyệt vời! Bạn đã thăng cấp ĐỒNG 🥉</h2>
                <p class="points-text">+ 200 ĐIỂM</p>
                <p style="font-size: 24px; color: #fff; margin-bottom: 15px;">Thời gian hoàn thành: ${timeTaken} giây</p>
                <div class="rank-bar" style="width:400px; height:30px; background:#333; border:3px solid #FFD700; border-radius:15px; overflow:hidden; margin:15px 0; box-shadow: 0 0 15px rgba(255, 215, 0, 0.5);">
                    <div style="width: 100%; height:100%; background:linear-gradient(90deg, #FFD700, #FFF);"></div>
                </div>
                <p style="font-size: 26px; color: #FFD700; font-weight: bold;">Tổng điểm: 1000 / 1000</p>
                <button class="action-btn" style="margin-top: 25px;" onclick="window.location.href='lv4.html'">CẤP ĐỘ TIẾP THEO ➔</button>
            `;
            document.getElementById('win-modal').classList.add('active');
        }

        function handleLose() { 
            gameActive = false; 
            clearTimeout(spawnTimeout); 
            bgMusic.pause();
            const meteors = document.getElementsByClassName('meteor');
            while(meteors[0]) meteors[0].parentNode.removeChild(meteors[0]);
            document.getElementById('lose-modal').classList.add('active'); 
        }

        window.addEventListener('keydown', function(event) {
            if (event.key === 't' || event.key === 'T') { handleWin(); }
            if (event.key === 'l' || event.key === 'L') { handleLose(); }
        });
